# Input Item

We have 3 types of inout items:

1. Defaul input item that is used in forms.
2. Input item when validation fails.
3. Input item used in the dashboard section.

export const Component = (args) => (
  <InputItem>
    <label>{args.text}</label>
    <input type="text" placeholder={args.placeholder} name="name" />
  </InputItem>
);

export const ComponentFail = (args) => (
  <InputItem fail>
    <label>{args.text}</label>
    <input type="text" placeholder={args.placeholder} name="name" />
  </InputItem>
);

export const ComponentExpand = (args) => (
  <InputItem expand>
    <label>{args.text}</label>
    <input type="text" placeholder={args.placeholder} name="name" />
  </InputItem>
);
